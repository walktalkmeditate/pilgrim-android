### Task 9 — VM plumbing: StateFlow + pin/unpin methods

**Files:**
- `app/src/main/java/org/walktalkmeditate/pilgrim/ui/walk/WalkSummaryViewModel.kt` (modify)

**Changes:**

3. Add:
   ```kotlin
   fun pinPhotos(uris: List<Uri>) {
       if (uris.isEmpty()) return
       viewModelScope.launch(Dispatchers.IO) {
           val now = System.currentTimeMillis()
           val refs = uris.map { uri ->
               runCatching {
                   context.contentResolver.takePersistableUriPermission(
                       uri, Intent.FLAG_GRANT_READ_URI_PERMISSION,
                   )
               }.onFailure {
                   android.util.Log.w(TAG, "takePersistableUriPermission failed for $uri", it)
               }
               PhotoPinRef(uri = uri.toString(), takenAt = readDateTaken(uri))
           }
           try {
               repository.pinPhotos(walkId, refs, pinnedAt = now)
           } catch (ce: CancellationException) { throw ce }
             catch (t: Throwable) {
               android.util.Log.e(TAG, "pinPhotos failed for walk $walkId", t)
           }
       }
   }

   fun unpinPhoto(photo: WalkPhoto) {
       viewModelScope.launch(Dispatchers.IO) {
           try {
               repository.unpinPhoto(photo.id)
           } catch (ce: CancellationException) { throw ce }
             catch (t: Throwable) {
               android.util.Log.e(TAG, "unpinPhoto failed id=${photo.id}", t)
           }
       }
   }

   private fun readDateTaken(uri: Uri): Long? = runCatching {
       context.contentResolver.query(
           uri,
           arrayOf(MediaStore.MediaColumns.DATE_TAKEN),
           null, null, null,
       )?.use { cursor ->
           if (cursor.moveToFirst()) {
               val col = cursor.getColumnIndex(MediaStore.MediaColumns.DATE_TAKEN)
               if (col >= 0 && !cursor.isNull(col)) cursor.getLong(col) else null
           } else null
       }
   }.getOrNull()
   ```

### Task 11 — `PhotoReliquarySection` composable

**Files:**
- `app/src/main/java/org/walktalkmeditate/pilgrim/ui/walk/reliquary/PhotoReliquarySection.kt` (new)
- `app/src/main/java/org/walktalkmeditate/pilgrim/ui/walk/reliquary/ReliquaryConstants.kt` (new) — `const val MAX_PINS_PER_WALK = 20`

**Composable shape:**
```kotlin
@Composable
fun PhotoReliquarySection(
    photos: List<WalkPhoto>,
    onPinPhotos: (List<Uri>) -> Unit,
    onUnpinPhoto: (WalkPhoto) -> Unit,
    modifier: Modifier = Modifier,
) {
    val slots = (MAX_PINS_PER_WALK - photos.size).coerceAtLeast(0)
    val multiLauncher = rememberLauncherForActivityResult(
        PickMultipleVisualMedia(maxItems = slots.coerceAtLeast(2))
    ) { uris -> onPinPhotos(uris) }
    val singleLauncher = rememberLauncherForActivityResult(
        PickVisualMedia()
    ) { uri -> onPinPhotos(listOfNotNull(uri)) }

    Column(modifier = modifier) {
        // Header row: title + Add button
        Row(...) {
            Text("Reliquary", ...)
            OutlinedButton(
                enabled = slots > 0,
                onClick = {
                    val req = PickVisualMediaRequest(
                        PickVisualMedia.ImageOnly
                    )
                    when {
                        slots == 1 -> singleLauncher.launch(req)
                        slots >= 2 -> multiLauncher.launch(req)
                    }
                },
            ) {
                Icon(Icons.Default.AddAPhoto, contentDescription = null)
                Spacer(Modifier.width(6.dp))
                Text(if (slots > 0) "Add" else "Full")
            }
        }
        if (photos.isNotEmpty()) {
            Spacer(Modifier.height(8.dp))
            PhotoGrid(photos = photos, onLongPress = { photoToRemove = it })
        }
    }
    // Confirmation dialog
    photoToRemove?.let { target ->
        AlertDialog(
            onDismissRequest = { photoToRemove = null },
            title = { Text("Remove from walk?") },
            text = { Text("This photo will be unpinned. Your photo library isn't changed.") },
            confirmButton = { TextButton(onClick = { onUnpinPhoto(target); photoToRemove = null }) { Text("Remove") } },
            dismissButton = { TextButton(onClick = { photoToRemove = null }) { Text("Keep") } },
        )
    }
}

@Composable
private fun PhotoGrid(
    photos: List<WalkPhoto>,
    onLongPress: (WalkPhoto) -> Unit,
) {
    // Non-lazy 3-col grid (chunked Rows) — avoids LazyVerticalGrid inside verticalScroll
    val cols = 3
    val rows = photos.chunked(cols)
    val haptics = LocalHapticFeedback.current
    Column(verticalArrangement = Arrangement.spacedBy(4.dp)) {
        rows.forEach { row ->
            Row(horizontalArrangement = Arrangement.spacedBy(4.dp)) {
                row.forEach { photo ->
                    val action = {
                        haptics.performHapticFeedback(HapticFeedbackType.LongPress)
                        onLongPress(photo)
                    }
                    Box(
                        modifier = Modifier
                            .weight(1f)
                            .aspectRatio(1f)
                            .clip(RoundedCornerShape(6.dp))
                            .background(pilgrimColors.parchmentSecondary)
                            .pointerInput(photo.id) {
                                detectTapGestures(onLongPress = { action() })
                            }
                            .semantics {
                                customActions = listOf(
                                    CustomAccessibilityAction("Remove from walk") {
                                        onLongPress(photo); true
                                    }
                                )
                            },
                    ) {
                        AsyncImage(
                            model = Uri.parse(photo.photoUri),
                            contentDescription = "Photo from this walk",
                            contentScale = ContentScale.Crop,
                            modifier = Modifier.fillMaxSize(),
                        )
                    }
                }
                // Pad trailing cells to keep columns aligned
                repeat(cols - row.size) {
                    Spacer(modifier = Modifier.weight(1f))
                }
            }
        }
    }
}
```
